def greet_in_multiple_languages(name):
	greetings = {
		'English' :f"Hello,{name}!"
		'Spanish' :f"Hola,{name}!"
		'French' :f"Bonjour,{name}!"
		'German' :f"Hallo,{name}!"
		'italian' :f"Ciao,{name}!",
}

for language, greeting in greetings.items():
	print(greeting)

name = input("what is your name?")
greet_in_multiple_languages(name)
